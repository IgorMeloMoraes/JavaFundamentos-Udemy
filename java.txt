package fundamentos;

public class PrimeiroPrograma {
	
	/* ATALHOS TECLADO: 
	 * main + CTRL + space: Ele prcura complementar o codigo que foi iniciado  
	 * syso + CTRL + space: Gera uma saida de texto, um output  
	 * CTRL + M: Abre e fecha a barra lateral de pastas
	 * CTRL + SHIFT + F11: Roda o codigo
	 * CTRL + S: Salva o projeto
	 * */
	
	/* COMO UM PROGRAMA JAVA É ORGANIZADO:
	 * Como organizamos um livro: Podemos sair do macro para o micro, pensando que um livro pode ser organizado por capitulos, que é organizado por paragrafos, que são organizados em palavras, em letras... 
	 * Ele é organizado em sentenças de codigo que terminam com ponto e virgula (;)
	 * O que vai indicar uma sequencia de codigo não é a quantidade de linhas e sim um ponto e virgula 
	 * O uso de ponto e virgula é obrigatorio
	 * Podemos quebrar o codigo em qualquer ponto? nao, podemos quebrar em alguns pontos especificos
	 * Utilizamos frequentemente pares de chaves
	 * As chaves significam blocos de codigo 
	 * serve para agrupar sentenças de codigo
	 * representa um paragrafo  */
	
	/* MAIN 
	 * É a porta de entrada do seu codigo java
	 * É um metodo 
	 * É apatir dele que vamos começar a executar o codigo
	 * Só se começa um promgrama a partir dele 
	 * é o inicio do programa
	 *   */
	public static void main(String[] args) {
		/*System: Sistema
		 * Out: Output / Saida
		 * Println: Imprima na linha
		 * -> Imprima na saida do sistema o seguinte*/
		System.out.println("Hello Word!");
		System.out.println("Ola mundo");
		
		
	}
	
	
}
